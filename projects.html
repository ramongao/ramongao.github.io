Wildfire Drone Coverage Simulation

Real-time simulation showing drone swarm tracking fire spread with collision avoidance

A Pygame simulation modeling drone swarm coverage of spreading forest fires with spatial separation constraints and real-time fire propagation.

Key Features:

  • Real-time fire spread using cellular automata
  • Multi-drone coordination with collision avoidance
  • Frontier-based target assignment algorithm
  • Vector-based movement and repulsion forces
  • Visual feedback system with coverage visualization

Python/Pygame Code

import pygame
import random
import math

pygame.init()

# ================= SCREEN =================
SCREEN_WIDTH = 1300
SCREEN_HEIGHT = 800
CELL_SIZE = 6

GRID_WIDTH = SCREEN_WIDTH // CELL_SIZE
GRID_HEIGHT = SCREEN_HEIGHT // CELL_SIZE

screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
pygame.display.set_caption("Forest Fire Drone Coverage")

clock = pygame.time.Clock()

# ================= COLORS =================
GREEN = (34, 139, 34)
RED = (255, 70, 70)
DARK_RED = (120, 0, 0)
BLUE = (30, 144, 255)
LIGHT_BLUE = (135, 206, 250)

# ================= FIRE GRID =================
grid = [[0 for _ in range(GRID_HEIGHT)] for _ in range(GRID_WIDTH)]
burn_time = [[0 for _ in range(GRID_HEIGHT)] for _ in range(GRID_WIDTH)]

# Initial fire
ix = random.randint(20, GRID_WIDTH - 20)
iy = random.randint(20, GRID_HEIGHT - 20)
grid[ix][iy] = 1

FIRE_SPREAD_CHANCE = 0.08
BURN_DURATION = 80
SPREAD_INTERVAL = 6
frame_count = 0

# ================= DRONES =================
NUM_DRONES = 5
DRONE_SPEED = 2.2
VIEW_RADIUS = 150
MIN_DRONE_DISTANCE = 160
REPULSION_STRENGTH = 300

class Drone:
    def __init__(self, idx):
        self.id = idx
        self.x = random.randint(0, SCREEN_WIDTH)
        self.y = random.randint(0, SCREEN_HEIGHT)

    def update(self, frontier, drones):
        fx, fy = 0, 0

        # Assign spatially separated frontier targets
        if frontier:
            target = frontier[(self.id * len(frontier)) // NUM_DRONES]
            tx, ty = target

            dx = tx - self.x
            dy = ty - self.y
            dist = math.hypot(dx, dy)

            if dist > VIEW_RADIUS * 0.7:
                fx += dx / dist
                fy += dy / dist

        # HARD separation enforcement
        for d in drones:
            if d is self:
                continue
            dx = self.x - d.x
            dy = self.y - d.y
            dist = math.hypot(dx, dy)

            if dist < MIN_DRONE_DISTANCE and dist > 0:
                strength = REPULSION_STRENGTH / (dist ** 2)
                fx += dx * strength
                fy += dy * strength

        # Move
        mag = math.hypot(fx, fy)
        if mag > 0:
            self.x += DRONE_SPEED * fx / mag
            self.y += DRONE_SPEED * fy / mag

        self.x = max(0, min(SCREEN_WIDTH, self.x))
        self.y = max(0, min(SCREEN_HEIGHT, self.y))

    def draw(self):
        pygame.draw.circle(
            screen, LIGHT_BLUE,
            (int(self.x), int(self.y)),
            VIEW_RADIUS, 1
        )
        pygame.draw.circle(screen, BLUE, (int(self.x), int(self.y)), 6)

# Create drones
drones = [Drone(i) for i in range(NUM_DRONES)]

# Fire update function
def update_fire():
    global frame_count
    frame_count += 1

    frontier = []

    for x in range(GRID_WIDTH):
        for y in range(GRID_HEIGHT):
            if grid[x][y] == 1:
                burn_time[x][y] += 1
                frontier.append((x * CELL_SIZE, y * CELL_SIZE))

                # Spread only every SPREAD_INTERVAL frames
                if frame_count % SPREAD_INTERVAL == 0:
                    for dx, dy in [(-1,0),(1,0),(0,-1),(0,1)]:
                        nx, ny = x + dx, y + dy
                        if 0 <= nx < GRID_WIDTH and 0 <= ny < GRID_HEIGHT:
                            if grid[nx][ny] == 0 and random.random() < FIRE_SPREAD_CHANCE:
                                grid[nx][ny] = 1

                if burn_time[x][y] > BURN_DURATION:
                    grid[x][y] = 2

    return frontier

# Main loop
run = True
while run:
    clock.tick(30)
    screen.fill(GREEN)

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            run = False

    frontier = update_fire()

    # Draw fire
    for x in range(GRID_WIDTH):
        for y in range(GRID_HEIGHT):
            if grid[x][y] == 1:
                pygame.draw.rect(
                    screen, RED,
                    (x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE)
                )
            elif grid[x][y] == 2:
                pygame.draw.rect(
                    screen, DARK_RED,
                    (x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE)
                )

    # Update drones
    for drone in drones:
        drone.update(frontier, drones)
        drone.draw()

    pygame.display.flip()

pygame.quit()

Supply Chain Management System

October 2025 - December 2025 Fordham University, New York

A comprehensive supply chain management system built with Python and MySQL for managing inventory, shipments, warehouses, and suppliers.

Key Features:

  • Automated database setup and configuration
  • Real-time inventory tracking across multiple warehouses
  • Transaction-safe shipment management
  • Low stock alerts and automated restocking
  • Interactive CLI interface for system management
Python MySQL Database Design JSON CRUD Operations 
import mysql.connector
import json
import os
import sys

class SupplyChain:
    def __init__(self):
        self.conn = None
        self.config = self.load_config()
        self.connect()
    
    def load_config(self):
        """Load or create configuration"""
        config_path = 'config.json'
        
        if not os.path.exists(config_path):
            config = {
                "database": {
                    "host": "localhost",
                    "name": "supply_chain",
                    "user": "root",
                    "password": ""
                }
            }
            with open(config_path, 'w') as f:
                json.dump(config, f, indent=2)
            print("Config file created")
            return config
        
        with open(config_path, 'r') as f:
            return json.load(f)
    
    def connect(self):
        """Connect to database"""
        db = self.config['database']
        host = db['host']
        user = db['user']
        password = db['password']
        db_name = db['name']
        
        # Try connection, ask for password if needed
        if not password:
            import getpass
            password = getpass.getpass("MySQL password (press Enter if none): ")
        
        try:
            # Try to connect to existing database
            self.conn = mysql.connector.connect(
                host=host,
                database=db_name,
                user=user,
                password=password,
                autocommit=False
            )
            print(f"Connected to {db_name}")
            
        except mysql.connector.Error as e:
            if e.errno == 1049:  # Database doesn't exist
                print(f"Database {db_name} not found. Creating...")
                self.create_database(host, user, password, db_name)
            else:
                print(f"Connection failed: {e}")
                return
    
    def create_database(self, host, user, password, db_name):
        """Create database and tables from SQL file"""
        try:
            # Connect without specifying database
            temp_conn = mysql.connector.connect(
                host=host,
                user=user,
                password=password
            )
            cursor = temp_conn.cursor()
            
            # Read and execute SQL file
            sql_path = 'schema.sql'
            
            with open(sql_path, 'r') as f:
                sql_content = f.read()
            
            # Execute each SQL statement
            statements = sql_content.split(';')
            for statement in statements:
                statement = statement.strip()
                if statement:  # Skip empty statements
                    try:
                        cursor.execute(statement)
                    except Exception as e:
                        print(f"Note: {e}")
            
            temp_conn.commit()
            cursor.close()
            temp_conn.close()
            
            # Connect to new database
            self.conn = mysql.connector.connect(
                host=host,
                database=db_name,
                user=user,
                password=password,
                autocommit=False
            )
            print(f"Database {db_name} created from SQL file")
            return True
            
        except Exception as e:
            print(f"Setup failed: {e}")
            return False
    
    def close(self):
        if self.conn:
            self.conn.close()
    
    def query(self, sql, params=None):
        """Execute query and return results"""
        cursor = self.conn.cursor(dictionary=True)
        cursor.execute(sql, params or ())
        result = cursor.fetchall()
        self.conn.commit()
        cursor.close()
        return result
    
    def execute(self, sql, params=None):
        """Execute update/insert query"""
        cursor = self.conn.cursor()
        cursor.execute(sql, params or ())
        self.conn.commit()
        cursor.close()
    
    # Core functions
    def create_shipment(self, from_id, to_id, product_id, quantity):
        """Create a shipment"""
        # Check stock
        stock = self.query(
            "SELECT quantity FROM inventory WHERE warehouse_id=%s AND product_id=%s",
            (from_id, product_id)
        )
        
        if not stock or stock[0]['quantity'] < quantity:
            print(f"Insufficient stock. Available: {stock[0]['quantity'] if stock else 0}")
            return False
        
        # Start transaction
        try:
            cursor = self.conn.cursor()
            cursor.execute("START TRANSACTION")
            
            # Create shipment
            cursor.execute(
                "INSERT INTO shipments (from_wh, to_wh, product_id, quantity) VALUES (%s, %s, %s, %s)",
                (from_id, to_id, product_id, quantity)
            )
            
            # Update inventory
            cursor.execute(
                "UPDATE inventory SET quantity=quantity-%s WHERE warehouse_id=%s AND product_id=%s",
                (quantity, from_id, product_id)
            )
            
            self.conn.commit()
            cursor.close()
            print("Shipment created")
            return True
            
        except Exception as e:
            self.conn.rollback()
            print(f"Failed: {e}")
            return False
    
    def deliver_shipment(self, shipment_id):
        """Mark shipment as delivered"""
        # Get shipment details
        shipment = self.query(
            "SELECT * FROM shipments WHERE shipment_id=%s AND status='pending'",
            (shipment_id,)
        )
        
        if not shipment:
            print("Shipment not found or already delivered")
            return False
        
        shipment = shipment[0]
        
        # Update shipment and inventory
        try:
            cursor = self.conn.cursor()
            cursor.execute("START TRANSACTION")
            
            # Update shipment status
            cursor.execute(
                "UPDATE shipments SET status='delivered' WHERE shipment_id=%s",
                (shipment_id,)
            )
            
            # Add to destination warehouse
            cursor.execute(
                """INSERT INTO inventory (warehouse_id, product_id, quantity) 
                   VALUES (%s, %s, %s) 
                   ON DUPLICATE KEY UPDATE quantity=quantity+VALUES(quantity)""",
                (shipment['to_wh'], shipment['product_id'], shipment['quantity'])
            )
            
            self.conn.commit()
            cursor.close()
            print("Shipment delivered")
            return True
            
        except Exception as e:
            self.conn.rollback()
            print(f"Failed: {e}")
            return False
    
    def check_low_stock(self):
        """Check for low stock items"""
        results = self.query("""
            SELECT p.name, COALESCE(SUM(i.quantity), 0) as total, p.min_stock
            FROM products p
            LEFT JOIN inventory i ON p.product_id=i.product_id
            GROUP BY p.product_id
            HAVING total < p.min_stock
        """)
        
        if results:
            print("Low stock items:")
            for r in results:
                print(f"{r['name']}: {r['total']} (min: {r['min_stock']})")
        else:
            print("All items stocked")
        
        return results
    
    def view_inventory(self, warehouse_id=None):
        """View inventory"""
        if warehouse_id:
            results = self.query("""
                SELECT p.name, i.quantity, p.price
                FROM inventory i
                JOIN products p ON i.product_id=p.product_id
                WHERE i.warehouse_id=%s
            """, (warehouse_id,))
            print(f"Inventory for warehouse {warehouse_id}:")
        else:
            results = self.query("""
                SELECT w.location, p.name, SUM(i.quantity) as total
                FROM inventory i
                JOIN warehouses w ON i.warehouse_id=w.warehouse_id
                JOIN products p ON i.product_id=p.product_id
                GROUP BY w.location, p.name
            """)
            print("All inventory:")
        
        for r in results:
            if 'location' in r:
                print(f"{r['location']}: {r['name']} x{r['total']}")
            else:
                print(f"{r['name']}: {r['quantity']}")
        
        return results
    
    def view_shipments(self, status=None):
        """View shipments"""
        if status:
            results = self.query("""
                SELECT s.shipment_id, w1.location as from_loc, w2.location as to_loc,
                       p.name, s.quantity, s.status
                FROM shipments s
                JOIN warehouses w1 ON s.from_wh=w1.warehouse_id
                JOIN warehouses w2 ON s.to_wh=w2.warehouse_id
                JOIN products p ON s.product_id=p.product_id
                WHERE s.status=%s
                ORDER BY s.created DESC
                LIMIT 10
            """, (status,))
            print(f"Shipments ({status}):")
        else:
            results = self.query("""
                SELECT s.shipment_id, w1.location as from_loc, w2.location as to_loc,
                       p.name, s.quantity, s.status
                FROM shipments s
                JOIN warehouses w1 ON s.from_wh=w1.warehouse_id
                JOIN warehouses w2 ON s.to_wh=w2.warehouse_id
                JOIN products p ON s.product_id=p.product_id
                ORDER BY s.created DESC
                LIMIT 10
            """)
            print("Recent shipments:")
        
        for s in results:
            print(f"#{s['shipment_id']}: {s['name']} x{s['quantity']} {s['from_loc']}->{s['to_loc']} ({s['status']})")
        
        return results
    
    def restock(self, warehouse_id, product_id, quantity):
        """Add stock to inventory"""
        try:
            self.execute(
                """INSERT INTO inventory (warehouse_id, product_id, quantity) 
                   VALUES (%s, %s, %s) 
                   ON DUPLICATE KEY UPDATE quantity=quantity+VALUES(quantity)""",
                (warehouse_id, product_id, quantity)
            )
            print(f"Added {quantity} units")
            return True
        except Exception as e:
            print(f"Failed: {e}")
            return False
    
    def get_warehouses(self):
        return self.query("SELECT warehouse_id, location FROM warehouses")
    
    def get_products(self):
        return self.query("SELECT product_id, name, price, min_stock FROM products")

def main():
    print("\nSupply Chain System")
    print("=" * 40)
    
    sc = SupplyChain()
    if not sc.conn:
        return
    
    while True:
        print("\n1. View inventory")
        print("2. View shipments")
        print("3. Create shipment")
        print("4. Deliver shipment")
        print("5. Check low stock")
        print("6. Restock")
        print("7. View products")
        print("8. View warehouses")
        print("9. Exit")
        
        choice = input("\nChoice: ")
        
        if choice == '1':
            wh = input("Warehouse ID (Enter for all): ")
            sc.view_inventory(int(wh) if wh else None)
        elif choice == '2':
            status = input("Status filter (pending/delivered or Enter): ")
            sc.view_shipments(status if status else None)
        elif choice == '3':
            frm = input("From warehouse: ")
            to = input("To warehouse: ")
            prod = input("Product: ")
            qty = input("Quantity: ")
            sc.create_shipment(int(frm), int(to), int(prod), int(qty))
        elif choice == '4':
            sid = input("Shipment ID: ")
            sc.deliver_shipment(int(sid))
        elif choice == '5':
            sc.check_low_stock()
        elif choice == '6':
            wh = input("Warehouse: ")
            prod = input("Product: ")
            qty = input("Quantity: ")
            sc.restock(int(wh), int(prod), int(qty))

        elif choice == '7':
            products = sc.get_products()
            print("\nProducts:")
            for p in products:
                print(f"ID: {p['product_id']} - {p['name']}")
                print(f"  Price: ${p['price']}")
                if 'min_stock' in p:
                    print(f"  Min stock: {p['min_stock']}")
                if 'supplier_id' in p:
                    print(f"  Supplier ID: {p['supplier_id']}")

        elif choice == '8':
            warehouses = sc.get_warehouses()
            print("\nWarehouses:")
            for w in warehouses:
                print(f"ID: {w['warehouse_id']} - {w['location']}")
        elif choice == '9':
            break
    
    sc.close()

if __name__ == "__main__":
    main()

SET FOREIGN_KEY_CHECKS = 0;
CREATE DATABASE IF NOT EXISTS supply_chain;
USE supply_chain;

CREATE TABLE suppliers (
    supplier_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    contact VARCHAR(100)
);

CREATE TABLE products (
    product_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    supplier_id INT,
    price DECIMAL(10,2),
    min_stock INT DEFAULT 10,
    FOREIGN KEY (supplier_id) REFERENCES suppliers(supplier_id)
);

CREATE TABLE warehouses (
    warehouse_id INT PRIMARY KEY AUTO_INCREMENT,
    location VARCHAR(100) NOT NULL
);

CREATE TABLE inventory (
    warehouse_id INT,
    product_id INT,
    quantity INT DEFAULT 0,
    PRIMARY KEY (warehouse_id, product_id),
    FOREIGN KEY (warehouse_id) REFERENCES warehouses(warehouse_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);

CREATE TABLE shipments (
    shipment_id INT PRIMARY KEY AUTO_INCREMENT,
    from_wh INT,
    to_wh INT,
    product_id INT,
    quantity INT,
    status VARCHAR(20) DEFAULT 'pending',
    created TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (from_wh) REFERENCES warehouses(warehouse_id),
    FOREIGN KEY (to_wh) REFERENCES warehouses(warehouse_id),
    FOREIGN KEY (product_id) REFERENCES products(product_id)
);
