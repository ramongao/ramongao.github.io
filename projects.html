import pygame
import random
import math

pygame.init()

# Screen configuration
SCREEN_WIDTH = 1300
SCREEN_HEIGHT = 800
CELL_SIZE = 6

GRID_WIDTH = SCREEN_WIDTH // CELL_SIZE
GRID_HEIGHT = SCREEN_HEIGHT // CELL_SIZE

screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT))
pygame.display.set_caption("Forest Fire Drone Coverage")

# Colors
GREEN = (34, 139, 34)
RED = (255, 70, 70)
DARK_RED = (120, 0, 0)
BLUE = (30, 144, 255)
LIGHT_BLUE = (135, 206, 250)

# Fire grid initialization
grid = [[0 for _ in range(GRID_HEIGHT)] for _ in range(GRID_WIDTH)]
burn_time = [[0 for _ in range(GRID_HEIGHT)] for _ in range(GRID_WIDTH)]

# Initial fire
ix = random.randint(20, GRID_WIDTH - 20)
iy = random.randint(20, GRID_HEIGHT - 20)
grid[ix][iy] = 1

FIRE_SPREAD_CHANCE = 0.08
BURN_DURATION = 80
SPREAD_INTERVAL = 6
frame_count = 0

# Drone class
class Drone:
    def __init__(self, idx):
        self.id = idx
        self.x = random.randint(0, SCREEN_WIDTH)
        self.y = random.randint(0, SCREEN_HEIGHT)

    def update(self, frontier, drones):
        fx, fy = 0, 0

        # Target assignment
        if frontier:
            target = frontier[(self.id * len(frontier)) // NUM_DRONES]
            tx, ty = target

            dx = tx - self.x
            dy = ty - self.y
            dist = math.hypot(dx, dy)

            if dist > VIEW_RADIUS * 0.7:
                fx += dx / dist
                fy += dy / dist

        # Collision avoidance
        for d in drones:
            if d is self:
                continue
            dx = self.x - d.x
            dy = self.y - d.y
            dist = math.hypot(dx, dy)

            if dist < MIN_DRONE_DISTANCE and dist > 0:
                strength = REPULSION_STRENGTH / (dist ** 2)
                fx += dx * strength
                fy += dy * strength

        # Movement
        mag = math.hypot(fx, fy)
        if mag > 0:
            self.x += DRONE_SPEED * fx / mag
            self.y += DRONE_SPEED * fy / mag

        self.x = max(0, min(SCREEN_WIDTH, self.x))
        self.y = max(0, min(SCREEN_HEIGHT, self.y))

    def draw(self):
        pygame.draw.circle(
            screen, LIGHT_BLUE,
            (int(self.x), int(self.y)),
            VIEW_RADIUS, 1
        )
        pygame.draw.circle(screen, BLUE, (int(self.x), int(self.y)), 6)

# Create drones
NUM_DRONES = 5
DRONE_SPEED = 2.2
VIEW_RADIUS = 150
MIN_DRONE_DISTANCE = 160
REPULSION_STRENGTH = 300

drones = [Drone(i) for i in range(NUM_DRONES)]

# Fire update function
def update_fire():
    global frame_count
    frame_count += 1
    frontier = []

    for x in range(GRID_WIDTH):
        for y in range(GRID_HEIGHT):
            if grid[x][y] == 1:
                burn_time[x][y] += 1
                frontier.append((x * CELL_SIZE, y * CELL_SIZE))

                if frame_count % SPREAD_INTERVAL == 0:
                    for dx, dy in [(-1,0),(1,0),(0,-1),(0,1)]:
                        nx, ny = x + dx, y + dy
                        if 0 <= nx < GRID_WIDTH and 0 <= ny < GRID_HEIGHT:
                            if grid[nx][ny] == 0 and random.random() < FIRE_SPREAD_CHANCE:
                                grid[nx][ny] = 1

                if burn_time[x][y] > BURN_DURATION:
                    grid[x][y] = 2

    return frontier

# Main game loop
clock = pygame.time.Clock()
run = True

while run:
    clock.tick(30)
    screen.fill(GREEN)

    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            run = False

    frontier = update_fire()

    # Draw fire
    for x in range(GRID_WIDTH):
        for y in range(GRID_HEIGHT):
            if grid[x][y] == 1:
                pygame.draw.rect(
                    screen, RED,
                    (x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE)
                )
            elif grid[x][y] == 2:
                pygame.draw.rect(
                    screen, DARK_RED,
                    (x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE)
                )

    # Update and draw drones
    for drone in drones:
        drone.update(frontier, drones)
        drone.draw()

    pygame.display.flip()

pygame.quit()
                

# drone.py - Enhanced Drone class
import math
import random

class Drone:
    def __init__(self, idx, x=None, y=None):
        self.id = idx
        self.x = x if x is not None else random.randint(0, SCREEN_WIDTH)
        self.y = y if y is not None else random.randint(0, SCREEN_HEIGHT)
        self.target = None
        self.history = []  # Path history for visualization
        self.coverage = 0  # Area coverage counter
        
    def update(self, frontier, drones):
        # Combined forces: attraction to fire + repulsion from drones
        fx, fy = 0, 0
        
        # 1. Fire tracking force
        if frontier:
            # Assign to closest frontier point
            self.target = min(frontier, 
                key=lambda p: math.dist((self.x, self.y), p))
            tx, ty = self.target
            
            dx = tx - self.x
            dy = ty - self.y
            dist = math.hypot(dx, dy)
            
            if dist > 10:  # Don't get too close
                attraction = min(1.0, VIEW_RADIUS / dist)
                fx += dx * attraction
                fy += dy * attraction
        
        # 2. Drone separation force
        for drone in drones:
            if drone.id == self.id:
                continue
                
            dx = self.x - drone.x
            dy = self.y - drone.y
            dist = math.hypot(dx, dy)
            
            if dist < MIN_DRONE_DISTANCE:
                # Strong repulsion when too close
                repulsion = REPULSION_STRENGTH / (dist ** 2 + 1)
                fx += dx * repulsion
                fy += dy * repulsion
        
        # 3. Boundary avoidance
        boundary_margin = 50
        if self.x < boundary_margin:
            fx += 1
        elif self.x > SCREEN_WIDTH - boundary_margin:
            fx -= 1
        if self.y < boundary_margin:
            fy += 1
        elif self.y > SCREEN_HEIGHT - boundary_margin:
            fy -= 1
        
        # Normalize and apply movement
        magnitude = math.hypot(fx, fy)
        if magnitude > 0:
            self.x += (fx / magnitude) * DRONE_SPEED
            self.y += (fy / magnitude) * DRONE_SPEED
        
        # Keep within bounds
        self.x = max(0, min(SCREEN_WIDTH, self.x))
        self.y = max(0, min(SCREEN_HEIGHT, self.y))
        
        # Record path (limit history length)
        self.history.append((self.x, self.y))
        if len(self.history) > 50:
            self.history.pop(0)
    
    def draw(self, screen):
        # Draw coverage radius
        pygame.draw.circle(screen, (135, 206, 250, 50), 
                          (int(self.x), int(self.y)), VIEW_RADIUS, 1)
        
        # Draw drone
        pygame.draw.circle(screen, (30, 144, 255), 
                          (int(self.x), int(self.y)), 8)
        
        # Draw drone ID
        font = pygame.font.Font(None, 20)
        text = font.render(str(self.id), True, (255, 255, 255))
        screen.blit(text, (self.x - 5, self.y - 8))
        
        # Draw path history
        if len(self.history) > 1:
            pygame.draw.lines(screen, (30, 144, 255, 100), 
                             False, self.history, 2)
